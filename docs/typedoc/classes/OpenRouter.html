    Class OpenRouter

    OpenRouter has capble to integrate with any framework. This router supports serverless functions as well as any server framework like NextJs Provides a simple way to create OData API routes in Next.js API routes.

    Example

    // In pages/api/users.ts
    import { OpenRouter } from '@phrasecode/odata';

    const router = new OpenRouter({
      dataSource: dataSource
    });

    export default async function handler(
      req: NextApiRequest,
      res: NextApiResponse
    ) {
      const path = req.url; // /api/users?$select=name&$filter=age gt 18
      const response = await router.queryable(User)(path);
      res.status(200).json(response);
    }

    // Now you can access: GET /api/users?$select=name&$filter=age gt 18
