    Class Model<T>

    Base Model class for defining OData entities. Extend this class and use decorators (@Table, @Column, @HasMany, etc.) to define your data models.

    Example

    @Table({
      tableName: 'users',
      underscored: true,
      timestamps: true
    })
    class User extends Model<User> {
      @Column({
        dataType: DataTypes.INTEGER,
        isPrimaryKey: true,
        isAutoIncrement: true
      })
      id: number;

      @Column({ dataType: DataTypes.STRING, isNullable: false })
      name: string;

      @Column({ dataType: DataTypes.STRING, isUnique: true })
      email: string;

      @HasMany(() => Order, {
        relation: [{ foreignKey: 'userId', sourceKey: 'id' }]
      })
      orders: Order[];
    }
